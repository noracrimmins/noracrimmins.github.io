library(tidyverse)
library(gt)
library(ggalt)
library(ggrepel)

international <- c("Giannis Antetokounmpo","Luka Dončić", "Joel Embiid", "Nikola Jokić", "Lauri Markkanen")

domestic <- c("LeBron James", "Stephen Curry", "Donovan Mitchell", "Damian Lillard", "Jayson Tatum")

nbaszn23advstats <- read_csv("23statsadv.csv")|>
  mutate(Season = 2023) |> 
  mutate(PlayerSeason = paste(Player, Season))

nbaszn22advstats <- read_csv("22nbaadvstats.csv") |> 
  mutate(Season = 2022) |> 
  mutate(PlayerSeason = paste(Player, Season))

nbaszn21advstats <- read_csv("21nbastatsadv.csv") |> 
  mutate(Season = 2021) |>
  mutate(PlayerSeason = paste(Player, Season))

nbaszn20advstats <- read_csv("20nbaadvstats.csv")|> 
  mutate(Season = 2020) |> 
  mutate(PlayerSeason = paste(Player, Season))

nbaszn19advstats <- read_csv("19nbaadvstats.csv") |> 
  mutate(Season = 2019) |> 
  mutate(PlayerSeason = paste(Player, Season))

nbaadvstats <- bind_rows(nbaszn19advstats, nbaszn20advstats, nbaszn21advstats, nbaszn22advstats, nbaszn23advstats)

topPER <- nbaadvstats |>
  filter(MP >2000) |>
  top_n(20, wt= PER) 

internationalPER <- topPER |>
  filter(Player %in% international)

domesticPER <- topPER |>
  filter(Player %in% domestic)

nbaszn23 <- read_csv("nbaregszn23.csv") |> mutate(Season = 2023) |> mutate(PlayerSeason = paste(Player, Season))

nbaszn22 <- read_csv("nbaregszn22.csv") |> mutate(Season = 2022) |> mutate(PlayerSeason = paste(Player, Season))

nbaszn21 <- read_csv("nbaregszn21.csv") |> mutate(Season = 2021) |> mutate(PlayerSeason = paste(Player, Season))

nbaszn20 <- read_csv("nbaregszn20.csv") |> mutate(Season = 2020) |> mutate(PlayerSeason = paste(Player, Season))

nbaszn19 <- read_csv("nbaregszn19.csv") |> mutate(Season = 2019)|> mutate(PlayerSeason = paste(Player, Season))

nbaszn <- bind_rows(nbaszn23, nbaszn22, nbaszn21, nbaszn20, nbaszn19)

internationalplayers <- nbaszn |> 
  filter(Player %in% international) |> 
  mutate(Type = "International")

domesticplayers <- nbaszn |> 
  filter(Player %in% domestic)|> 
  mutate(Type = "Domestic")

players <- bind_rows(internationalplayers, domesticplayers)

players |>
  group_by(Player, Type) |>
  summarise(
    G = sum(G),
    PTS = sum(PTS)
    ) |>
  mutate(ppg = PTS/G) |> 
  arrange(-ppg) |> 
  ungroup() |> 
  gt() |> 
  tab_header(
    title = "International and domestic players put up similar scoring numbers",
    subtitle = "The difference between these two groups go beyond scoring."
  ) |> tab_style(
    style = cell_text(color = "black", weight = "bold", align = "left"),
    locations = cells_title("title")
  ) |> tab_style(
    style = cell_text(color = "black", align = "left"),
    locations = cells_title("subtitle")
  ) |>
  tab_source_note(
    source_note = md("**By:** Nora Crimmins  |  **Source:** Source: Sports-Reference.com")
  ) |>
  tab_style(
     locations = cells_column_labels(columns = everything()),
     style = list(
       cell_borders(sides = "bottom", weight = px(3)),
       cell_text(weight = "bold", size=12)
     )
   ) |>
  opt_row_striping() |> 
  opt_table_lines("none")|>
  tab_style(
    style = list(
      cell_fill(color = "red"),
      cell_text(color = "white")
      ),
    locations = cells_body(
      columns = ppg)
  )
